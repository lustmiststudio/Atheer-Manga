    <script>
        /**
         * ATHEER MANGA FRAMEWORK v5.2.0
         * Lead Developer: Haitheem
         * Architecture: High-Performance API Integration
         */

        const ATHEER_ENV = {
            API_BASE: 'https://api.mangadex.org',
            CDN_BASE: 'https://uploads.mangadex.org',
            PAGE_LIMIT: 24,
            GLOBAL_OFFSET: 0,
            ACTIVE_MANGA_ID: null,
            ACTIVE_MANGA_TITLE: '',
            ACTIVE_CH_LIST: [],
            CURRENT_GENRE: null,
            SEARCH_TERM: '',
            IS_FETCHING: false
        };

        const AtheerApp = {
            
            // 1. App Bootstrapping
            async initialize() {
                try {
                    await Clerk.load();
                    this.handleAuthentication();
                    this.loadSystemData();
                    this.attachEventListeners();
                    this.checkUserHistory();
                } catch (err) {
                    console.error("Critical System Boot Failure:", err);
                    this.loadSystemData(); // Boot without auth if needed
                }
            },

            handleAuthentication() {
                const btn = document.getElementById('auth-trigger');
                if (Clerk.user) {
                    btn.style.display = 'none';
                    Clerk.mountUserButton(document.getElementById('clerk-user-mount'));
                } else {
                    btn.style.display = 'block';
                    btn.onclick = () => Clerk.openSignIn();
                }
            },

            async loadSystemData() {
                this.updateBootLog("Loading Categories...");
                await this.fetchGenres();
                
                this.updateBootLog("Fetching Latest Releases...");
                await this.fetchMangaData();

                // Finish Booting
                setTimeout(() => {
                    const bs = document.getElementById('boot-screen');
                    bs.style.opacity = '0';
                    setTimeout(() => bs.style.display = 'none', 600);
                }, 1500);
            },

            updateBootLog(msg) {
                document.getElementById('boot-log').innerText = msg;
            },

            // 2. Data Fetching Engine (CORE)
            async fetchMangaData(reset = false) {
                if (ATHEER_ENV.IS_FETCHING) return;
                ATHEER_ENV.IS_FETCHING = true;

                if (reset) {
                    ATHEER_ENV.GLOBAL_OFFSET = 0;
                    document.getElementById('main-grid').innerHTML = '';
                }

                document.getElementById('global-loader').style.display = 'block';

                let apiURL = `${ATHEER_ENV.API_BASE}/manga?limit=${ATHEER_ENV.PAGE_LIMIT}&offset=${ATHEER_ENV.GLOBAL_OFFSET}&includes[]=cover_art&contentRating[]=safe&contentRating[]=suggestive`;

                // Add Filters
                if (ATHEER_ENV.SEARCH_TERM) {
                    apiURL += `&title=${encodeURIComponent(ATHEER_ENV.SEARCH_TERM)}`;
                } else if (ATHEER_ENV.CURRENT_GENRE) {
                    apiURL += `&includedTags[]=${ATHEER_ENV.CURRENT_GENRE}`;
                } else {
                    apiURL += `&order[followedCount]=desc`;
                }

                try {
                    const response = await fetch(apiURL);
                    const result = await response.json();
                    this.renderMangaCards(result.data);
                    ATHEER_ENV.GLOBAL_OFFSET += ATHEER_ENV.PAGE_LIMIT;
                } catch (e) {
                    this.showNotification("خطأ في الاتصال بالخادم");
                } finally {
                    ATHEER_ENV.IS_FETCHING = false;
                    document.getElementById('global-loader').style.display = 'none';
                }
            },

            renderMangaCards(mangaList) {
                const grid = document.getElementById('main-grid');
                
                mangaList.forEach(manga => {
                    const mID = manga.id;
                    const mTitle = manga.attributes.title.en || Object.values(manga.attributes.title)[0] || 'Unknown';
                    const coverRel = manga.relationships.find(r => r.type === 'cover_art');
                    const coverFile = coverRel ? coverRel.attributes.fileName : '';
                    const finalCover = coverFile ? `${ATHEER_ENV.CDN_BASE}/covers/${mID}/${coverFile}.256.jpg` : 'https://via.placeholder.com/256x380?text=No+Cover';

                    const card = document.createElement('div');
                    card.className = 'manga-card fade-in';
                    card.innerHTML = `
                        <div class="cover-wrapper">
                            <img src="${finalCover}" class="cover" alt="${mTitle}" loading="lazy">
                        </div>
                        <div style="padding: 12px;">
                            <p style="font-size: 13px; font-weight: 700; height: 38px; overflow: hidden; line-height: 1.4;">${mTitle}</p>
                            <div style="display: flex; align-items: center; gap: 5px; margin-top: 8px;">
                                <span style="font-size: 9px; color: var(--accent-primary); border: 1px solid var(--accent-primary); padding: 1px 5px; border-radius: 3px;">MANGA</span>
                                <span style="font-size: 9px; opacity: 0.4;">MangaDex API</span>
                            </div>
                        </div>
                    `;
                    card.onclick = () => this.openMangaDetails(mID, mTitle);
                    grid.appendChild(card);
                });
            },

            // 3. Category System (FIXED)
            async fetchGenres() {
                try {
                    const res = await fetch(`${ATHEER_ENV.API_BASE}/manga/tag`);
                    const json = await res.json();
                    const box = document.getElementById('category-box');
                    
                    // Filter for specific genres or take the top ones
                    json.data.slice(0, 30).forEach(tag => {
                        const tagName = tag.attributes.name.en;
                        const tagID = tag.id;

                        const btn = document.createElement('div');
                        btn.className = 'nav-btn';
                        btn.innerHTML = `<span># ${tagName}</span>`;
                        btn.onclick = () => this.applyGenreFilter(tagID, tagName);
                        box.appendChild(btn);
                    });
                } catch (err) { console.error("Genre load error"); }
            },

            applyGenreFilter(id, name) {
                ATHEER_ENV.CURRENT_GENRE = id;
                ATHEER_ENV.SEARCH_TERM = '';
                document.getElementById('current-view-title').innerText = `تصنيف: ${name}`;
                toggleSidebar(false);
                this.fetchMangaData(true);
            },

            // 4. Random Discovery
            async getLucky() {
                toggleSidebar(false);
                this.showGlobalLoader(true);
                try {
                    const res = await fetch(`${ATHEER_ENV.API_BASE}/manga/random?includes[]=cover_art`);
                    const json = await res.json();
                    const m = json.data;
                    const title = m.attributes.title.en || Object.values(m.attributes.title)[0];
                    this.openMangaDetails(m.id, title);
                } catch (e) {
                    this.showNotification("حاول مرة أخرى...");
                } finally {
                    this.showGlobalLoader(false);
                }
            },

            // 5. Chapter & Reader Logic
            async openMangaDetails(id, title) {
                ATHEER_ENV.ACTIVE_MANGA_ID = id;
                ATHEER_ENV.ACTIVE_MANGA_TITLE = title;
                
                const modal = document.getElementById('details-modal');
                const list = document.getElementById('chapters-container');
                document.getElementById('modal-manga-name').innerText = title;
                
                modal.style.display = 'flex';
                list.innerHTML = '<div style="text-align:center; padding:40px;"><div class="loader-ring" style="display:inline-block; width:30px; height:30px;"></div></div>';

                try {
                    const res = await fetch(`${ATHEER_ENV.API_BASE}/manga/${id}/feed?translatedLanguage[]=ar&translatedLanguage[]=en&order[chapter]=desc&limit=500`);
                    const json = await res.json();
                    ATHEER_ENV.ACTIVE_CH_LIST = json.data;

                    if (ATHEER_ENV.ACTIVE_CH_LIST.length === 0) {
                        list.innerHTML = '<p style="text-align:center; opacity:0.5;">لا توجد فصول متوفرة بالعربية أو الإنجليزية حالياً.</p>';
                    } else {
                        list.innerHTML = ATHEER_ENV.ACTIVE_CH_LIST.map((ch, i) => `
                            <div class="nav-btn" onclick="AtheerApp.initiateReader(${i})" style="justify-content: space-between; background: rgba(255,255,255,0.02);">
                                <span>الفصل ${ch.attributes.chapter || '0'}</span>
                                <span style="font-size:10px; background:var(--accent-secondary); color:#fff; padding:2px 7px; border-radius:4px;">${ch.attributes.translatedLanguage.toUpperCase()}</span>
                            </div>
                        `).join('');
                    }
                } catch (err) {
                    list.innerHTML = '<p style="color:red; text-align:center;">خطأ في تحميل الفصول</p>';
                }
            },

            async initiateReader(idx) {
                const chapter = ATHEER_ENV.ACTIVE_CH_LIST[idx];
                const reader = document.getElementById('reader-overlay');
                const pBox = document.getElementById('reader-pages-box');
                
                closeModal();
                reader.style.display = 'flex';
                document.body.style.overflow = 'hidden';
                
                document.getElementById('reader-manga-title').innerText = ATHEER_ENV.ACTIVE_MANGA_TITLE;
                document.getElementById('reader-current-ch').innerText = `الفصل ${chapter.attributes.chapter}`;
                
                pBox.innerHTML = '<div style="text-align:center; padding:100px;"><div class="loader-ring" style="display:inline-block;"></div><p style="margin-top:20px;">جاري تحضير الصفحات...</p></div>';

                try {
                    const res = await fetch(`${ATHEER_ENV.API_BASE}/at-home/server/${chapter.id}`);
                    const json = await res.json();
                    const hash = json.chapter.hash;
                    const pageData = json.chapter.data;

                    pBox.innerHTML = pageData.map(img => `
                        <img src="${json.baseUrl}/data/${hash}/${img}" loading="lazy" alt="Manga Page">
                    `).join('');
                    
                    this.saveToHistory(ATHEER_ENV.ACTIVE_MANGA_ID, ATHEER_ENV.ACTIVE_MANGA_TITLE, idx, chapter.attributes.chapter);
                    pBox.scrollTop = 0;
                } catch (err) {
                    pBox.innerHTML = '<div style="text-align:center; padding:50px; color:red;">فشل تحميل محتوى الفصل.</div>';
                }
            },

            // 6. Support Systems
            saveToHistory(id, title, idx, label) {
                const log = { id, title, idx, label, time: Date.now() };
                localStorage.setItem('atheer_last_read', JSON.stringify(log));
                this.checkUserHistory();
            },

            checkUserHistory() {
                const raw = localStorage.getItem('atheer_last_read');
                if (raw) {
                    const data = JSON.parse(raw);
                    document.getElementById('history-section').style.display = 'block';
                    document.getElementById('hist-title').innerText = data.title;
                    document.getElementById('hist-ch').innerText = `استكمال الفصل ${data.label}`;
                    document.getElementById('resume-btn').onclick = () => {
                        this.openMangaDetails(data.id, data.title).then(() => this.initiateReader(data.idx));
                    };
                }
            },

            resetToHome() {
                ATHEER_ENV.CURRENT_GENRE = null;
                ATHEER_ENV.SEARCH_TERM = '';
                document.getElementById('current-view-title').innerText = 'أحدث الإضافات';
                document.getElementById('main-search').value = '';
                toggleSidebar(false);
                this.fetchMangaData(true);
            },

            attachEventListeners() {
                // Search Engine with Debounce
                let sTimer;
                document.getElementById('main-search').addEventListener('input', (e) => {
                    clearTimeout(sTimer);
                    sTimer = setTimeout(() => {
                        ATHEER_ENV.SEARCH_TERM = e.target.value;
                        ATHEER_ENV.CURRENT_GENRE = null;
                        this.fetchMangaData(true);
                    }, 800);
                });

                // Infinite Scroll Observer
                window.addEventListener('scroll', () => {
                    if ((window.innerHeight + window.scrollY) >= document.body.offsetHeight - 1500) {
                        this.fetchMangaData();
                    }
                });
            },

            closeReader() {
                document.getElementById('reader-overlay').style.display = 'none';
                document.body.style.overflow = 'auto';
            },

            reOpenDetailsFromReader() {
                this.closeReader();
                this.openMangaDetails(ATHEER_ENV.ACTIVE_MANGA_ID, ATHEER_ENV.ACTIVE_MANGA_TITLE);
            },

            showGlobalLoader(show) {
                document.getElementById('boot-screen').style.display = show ? 'flex' : 'none';
                document.getElementById('boot-screen').style.opacity = show ? '1' : '0';
            },

            showNotification(msg) {
                console.log("System Notice: " + msg);
            }
        };

        // UI Helpers
        function toggleSidebar(state) {
            document.getElementById('sidebar').classList.toggle('active', state);
            document.getElementById('sidebar-dimmer').classList.toggle('show', state);
        }

        function closeModal() {
            document.getElementById('details-modal').style.display = 'none';
        }

        // Start App
        document.addEventListener('DOMContentLoaded', () => AtheerApp.initialize());

    </script>
